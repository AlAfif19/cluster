# Test Database Setup

Options:
1. Use same database as development (easiest, but may interfere)
2. Create separate test database (cleaner isolation)
3. Use SQLite in-memory for tests (fastest, but MySQL-specific features won't work)

For v1, we'll use option 1 with transaction rollback for isolation:
- Tests run within database transactions
- Transactions are rolled back after each test
- Ensures tests don't pollute the database
- Works with MySQL-specific features

"""
Pytest configuration and fixtures for authentication tests.
"""

import pytest
from fastapi.testclient import TestClient
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker, Session
from sqlalchemy.pool import StaticPool

from backend.app.main import app
from backend.app.database import get_db, Base
from backend.app.models.user import User, get_password_hash


# Create test database engine (uses in-memory SQLite for speed)
# For MySQL tests, use a separate test database
TEST_DATABASE_URL = "sqlite:///:memory:"

test_engine = create_engine(
    TEST_DATABASE_URL,
    connect_args={"check_same_thread": False},
    poolclass=StaticPool,
)
TestingSessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=test_engine)


@pytest.fixture(scope="function")
def db_session():
    """
    Create a fresh database session for each test.
    All changes are rolled back after the test.
    """
    # Create tables
    Base.metadata.create_all(bind=test_engine)

    # Create session
    session = TestingSessionLocal()

    try:
        yield session
    finally:
        # Rollback and cleanup
        session.rollback()
        session.close()
        Base.metadata.drop_all(bind=test_engine)


@pytest.fixture(scope="function")
def client(db_session: Session):
    """
    Create a test client with database override.
    """
    def override_get_db():
        try:
            yield db_session
        finally:
            pass

    app.dependency_overrides[get_db] = override_get_db

    with TestClient(app) as test_client:
        yield test_client

    app.dependency_overrides.clear()


@pytest.fixture(scope="function")

"""
Tests for user registration endpoint.
"""

import pytest
from fastapi import status


def test_register_user(client):
    """Test successful user registration."""
    response = client.post(
        "/api/v1/users/register",
        json={
            "email": "newuser@example.com",
            "password": "newpass123",
            "full_name": "New User"
        }
    )

    assert response.status_code == status.HTTP_201_CREATED
    data = response.json()
    assert "id" in data
    assert data["email"] == "newuser@example.com"
    assert data["full_name"] == "New User"
    assert "password" not in data
    assert "hashed_password" not in data


def test_register_duplicate_email(client, test_user):
    """Test that duplicate email returns 400 error."""
    response = client.post(
        "/api/v1/users/register",
        json={
            "email": test_user.email,
            "password": "anotherpass123",
            "full_name": "Another User"
        }
    )

    assert response.status_code == status.HTTP_400_BAD_REQUEST
    assert "already registered" in response.json()["detail"].lower()


def test_register_invalid_email(client):
    """Test that invalid email format is rejected."""
    response = client.post(
        "/api/v1/users/register",
        json={
            "email": "not-an-email",
            "password": "testpass123",
            "full_name": "Test User"
        }
    )

    # Should return 422 (validation error)
    assert response.status_code == status.HTTP_422_UNPROCESSABLE_ENTITY


def test_register_short_password(client):
    """Test that short password is rejected."""
    response = client.post(
        "/api/v1/users/register",
        json={
            "email": "short@example.com",
            "password": "short",
            "full_name": "Short User"
        }
    )

    # Should return 422 (validation error)
    assert response.status_code == status.HTTP_422_UNPROCESSABLE_ENTITY


def test_list_users(client, test_user):

    assert response.status_code == status.HTTP_200_OK
    users = response.json()
    assert len(users) >= 1

"""
Tests for authentication endpoints (login, logout, protected routes).
"""

import pytest
from fastapi import status


def test_login_success(client, test_user):
    """Test successful login with valid credentials."""
    response = client.post(
        "/api/v1/auth/login",
        data={"username": test_user.email, "password": "testpass123"}
    )

    assert response.status_code == status.HTTP_200_OK
    data = response.json()
    assert "access_token" in data
    assert data["token_type"] == "bearer"
    assert len(data["access_token"]) > 0


def test_login_invalid_email(client):
    """Test login with non-existent email."""
    response = client.post(
        "/api/v1/auth/login",
        data={"username": "nonexistent@example.com", "password": "testpass123"}
    )

    assert response.status_code == status.HTTP_401_UNAUTHORIZED
    assert "incorrect" in response.json()["detail"].lower()


def test_login_invalid_password(client, test_user):
    """Test login with wrong password."""
    response = client.post(
        "/api/v1/auth/login",
        data={"username": test_user.email, "password": "wrongpassword"}
    )

    assert response.status_code == status.HTTP_401_UNAUTHORIZED
    assert "incorrect" in response.json()["detail"].lower()


def test_get_current_user(client, auth_headers):

    assert "password" not in data
