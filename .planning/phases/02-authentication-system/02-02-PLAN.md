   - From backend.app.database: get_db
   - From backend.app.models.user: User
   - From backend.app.schemas.user: UserCreate, User as UserSchema
   - From backend.app.core.security: get_password_hash, create_access_token, ACCESS_TOKEN_EXPIRE_MINUTES
   - From typing: List

2. Create router:
   - router = APIRouter()

3. Implement registration endpoint:
   @router.post("/register", response_model=UserSchema, status_code=status.HTTP_201_CREATED)
   def register_user(user: UserCreate, db: Session = Depends(get_db)):
       """
       Register a new user with email and password.
       """
       # Check if user with email already exists
       db_user = db.query(User).filter(User.email == user.email).first()
       if db_user:
           raise HTTPException(
               status_code=status.HTTP_400_BAD_REQUEST,
               detail="Email already registered"
           )

       # Hash password
       hashed_password = get_password_hash(user.password)

       # Create new user
       db_user = User(
           email=user.email,
           hashed_password=hashed_password,
           full_name=user.full_name
       )

       # Save to database
       db.add(db_user)
       db.commit()
       db.refresh(db_user)

       return db_user

4. Add helper endpoint to list all users (for development/testing):
   @router.get("/", response_model=List[UserSchema])
   def read_users(db: Session = Depends(get_db), skip: int = 0, limit: int = 100):
       """Get list of users (development only)"""
       users = db.query(User).offset(skip).limit(limit).all()
       return users

5. Add endpoint to get user by email (for login verification):
   @router.get("/email/{email}", response_model=UserSchema)
   def get_user_by_email(email: str, db: Session = Depends(get_db)):
       """Get user by email address"""
       user = db.query(User).filter(User.email == email).first()
       if not user:
           raise HTTPException(status_code=404, detail="User not found")
       return user
